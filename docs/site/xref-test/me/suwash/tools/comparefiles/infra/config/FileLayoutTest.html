43          // レコードリストが指定されていない場合
44          //------------------------------------------------------------------------------------------
45          FileLayout fileLayout = new FileLayout();
46          try {
47              fileLayout.getFirstRecordByteLength();
48              fail();
49          } catch (CompareFilesException e) {
50              log.trace("目検用", e);
51              assertEquals("メッセージコード", Const.CHECK_NOTNULL, e.getMessageId());
52          }
53          assertEquals("定義内容から、レコードパターン＝なし を返すこと", RecordPattern.None, fileLayout.getRecordPattern());
54  
55          //------------------------------------------------------------------------------------------
56          // レコードリストにヘッダー、データが指定されている場合
57          //------------------------------------------------------------------------------------------
58          RecordLayout headerRecord = new RecordLayout();
59          headerRecord.setType(RecordType.Header);
60          RecordLayout dataRecord = new RecordLayout();
61          dataRecord.setType(RecordType.Data);
62  
63          List<RecordLayout> recordList = new ArrayList<RecordLayout>();
64          recordList.add(headerRecord);
65          recordList.add(dataRecord);
66  
67          fileLayout.setRecordList(recordList);
68          assertEquals("定義内容から、レコードパターン＝HeaderData を返すこと", RecordPattern.HeaderData, fileLayout.getRecordPattern());
69  
70      }
