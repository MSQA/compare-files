12  /**
13   * アプリケーション層のLayerSuperType基底クラス。
14   *
15   * @param <I> 入力データモデル
16   * @param <O> 出力データモデル
17   */
18  @lombok.extern.slf4j.Slf4j
19  public abstract class BaseFacade<I extends Input, O extends Output<I>> extends BaseLayerSuperType<I, O> implements Facade<I, O> {
20  
21      /*
22       * (非 Javadoc)
23       * @see me.suwash.tools.comparefiles.infra.policy.BaseLayerSuperType#execute(me.suwash.ddd.policy.Input)
24       */
25      @Override
26      public O execute(I input) {
27          boolean hasError = false;
28          O output = null;
29          try {
30              output = super.execute(input);
31  
32          } catch (Exception e) {
33              hasError = true;
34              log.error(
35                  CompareFilesMessageSource.getInstance().getMessage(
36                      Const.ERRORHANDLE,
37                      new Object[] {this.getClass().getSimpleName(), e.getMessage()}));
38              for (final Repository curRepo : Context.getInstance().getRepositoryList()) {
39                  log.warn("  force-rollback: " + curRepo.toString());
40                  curRepo.rollback();
41              }
42              // 再throw
43              throw e;
44  
45          } finally {
46              if (!hasError) {
47                  for (final Repository curRepo : Context.getInstance().getRepositoryList()) {
48                      log.warn("  force-commit:" + curRepo.toString());
49                      curRepo.commit();
50                  }
51              }
52  
53          }
54  
55          return output;
56      }
57  
58  }
