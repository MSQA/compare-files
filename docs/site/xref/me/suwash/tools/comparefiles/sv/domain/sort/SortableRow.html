1   package me.suwash.tools.comparefiles.sv.domain.sort;
2   
3   import me.suwash.tools.comparefiles.sv.domain.BaseRow;
4   import me.suwash.util.CompareUtils;
5   
6   /**
7    * ソート用の行データ。
8    */
9   public class SortableRow extends BaseRow implements Comparable<SortableRow> {
10  
11      /*
12       * (非 Javadoc)
13       * @see java.lang.Comparable#compareTo(java.lang.Object)
14       */
15      @Override
16      public int compareTo(final SortableRow targetObject) {
17          // 相手がnullの場合、nullを小さいものとして扱う。
18          if (targetObject == null) {
19              return -1;
20          }
21  
22          // キーで比較
23          int result = CompareUtils.deepCompare(this.getKeyMap(), targetObject.getKeyMap());
24          if (result == 0) {
25              // キーが一致している場合は、その他の項目で比較
26              result = CompareUtils.deepCompare(this.getValueMap(), targetObject.getValueMap());
27          }
28          return result;
29      }
30  
31  }
