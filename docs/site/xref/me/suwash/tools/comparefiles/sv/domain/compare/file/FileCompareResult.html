110     /*
111      * (非 Javadoc)
112      * @see me.suwash.tools.comparefiles.sv.domain.compare.CompareResult#compare()
113      */
114     @Override
115     public void compare() {
116         // --------------------------------------------------------------------------------
117         // 事前処理
118         // --------------------------------------------------------------------------------
119         // 実行開始時刻
120         this.startTime = new Date();
121 
122         // ----------------------------------------
123         // 単項目チェック
124         // ----------------------------------------
125         ValidateUtils.validate(this.input);
126 
127         // ----------------------------------------
128         // 関連チェック
129         // ----------------------------------------
130         // 出力ディレクトリが存在しない場合、作成
131         final String outputDirPath = input.getOutputDirPath();
132         FileUtils.mkdirs(outputDirPath);
133 
134         // 出力ファイルが存在する場合、削除
135         final CompareFilesConfig systemConfig = input.getSystemConfig();
136         final String outputFileName = getOutputFileName(systemConfig);
137         final String outputFilePath = outputDirPath + "/" + outputFileName;
138         FileUtils.setupOverwrite(outputFilePath);
139 
140         // --------------------------------------------------------------------------------
141         // 本処理
142         // --------------------------------------------------------------------------------
143         // ----------------------------------------
144         // 設定
145         // ----------------------------------------
146         // 左ファイル
147         final String leftFilePath = input.getLeftFilePath();
148         final File leftFile = new File(leftFilePath);
149         this.leftFilePath = leftFile.getAbsolutePath();
150 
151         // 右ファイル
152         final String rightFilePath = input.getRightFilePath();
153         final File rightFile = new File(rightFilePath);
154         this.rightFilePath = rightFile.getAbsolutePath();
155 
156         // ----------------------------------------
157         // 共通比較
158         // ----------------------------------------
159         log.info("  ・ファイル比較 左：" + leftFile.getName() + "、右：" + rightFile.getName());
160 
161         // ファイル名除外ルール確認
162         log.info("    ・ファイル名除外ルール確認");
163         if (isIgnore(leftFile, systemConfig.getIgnoreFileRegexList())) {
164             this.status = CompareStatus.Ignore;
165             this.endTime = new Date();
166             return;
167         }
168 
169         // ファイルの存在比較
170         log.info("    ・ファイルの存在比較");
171         final CompareStatus fileExistCheckStatus = fileExistCheck(leftFile, rightFile);
172         if (!CompareStatus.OK.equals(fileExistCheckStatus)) {
173             this.status = fileExistCheckStatus;
174             this.endTime = new Date();
175             return;
176         }
177 
178         // ファイルレイアウト判定
179         log.info("    ・ファイルレイアウト判定");
180         final FileLayoutManager layoutManager = FileLayoutManager.getInstance();
181         final String leftFileName = leftFile.getName();
182         this.fileLayout = layoutManager.getLayout(leftFileName, systemConfig);
183 
184         // ----------------------------------------
185         // ファイル比較
186         // ----------------------------------------
187         // ファイルフォーマット判定
188         FileFormat fileFormat = null;
189         final String ext = getExt(leftFileName);
190         log.debug("      ・拡張子：" + ext);
191         if (this.fileLayout == null && ImageCompareUtils.isAllowedExt(ext)) {
192             fileFormat = FileFormat.Image;
193         }
194 
195         // ファイルフォーマットに合わせて比較実行
196         if (FileFormat.Image.equals(fileFormat)) {
197             // 画像比較
198             final ImageFileCompareResult result = new ImageFileCompareResult(
199                 this.leftFilePath,
200                 this.rightFilePath,
201                 this.fileLayout,
202                 outputDirPath,
203                 systemConfig,
204                 ext);
205             result.compare();
206             copyResult(result);
207 
208         } else {
209             // テキスト比較
210             final TextFileCompareResult result = new TextFileCompareResult(
211                 this.leftFilePath,
212                 this.rightFilePath,
213                 this.fileLayout,
214                 outputDirPath,
215                 systemConfig
216                 );
217             result.compare();
218             copyResult(result);
219         }
220 
221         // --------------------------------------------------------------------------------
222         // 事後処理
223         // --------------------------------------------------------------------------------
224         // workディレクトリの削除
225         if (systemConfig.isDeleteWorkDir()) {
226             final String workDirPath = outputDirPath + "/" + Const.DIRNAME_WORK;
227             FileUtils.rmdirs(workDirPath);
228         }
229     }

231     /**
232      * ファイル名から拡張子を返します。
233      *
234      * @param fileName ファイル名
235      * @return 拡張子
236      */
237     private String getExt(final String fileName) {
238         final String ext = fileName.substring(fileName.lastIndexOf('.') + 1);
239         // ドットが含まれていない場合、空文字を返す
240         if (fileName.equals(ext)) {
241             return StringUtils.EMPTY;
242         }
243         return ext;
244     }

246     /**
247      * 出力ファイル名 を返します。
248      *
249      * @param systemConfig システム設定
250      * @return 出力ファイル名
251      */
252     protected String getOutputFileName(final CompareFilesConfig systemConfig) {
253         final File leftFile = new File(input.getLeftFilePath());
254         final File rightFile = new File(input.getRightFilePath());
255 
256         final StringBuilder outputFileNameBuilder = new StringBuilder();
257         outputFileNameBuilder.append(systemConfig.getCompareDetailFilePrefix()).append(rightFile.getName());
258 
259         // --------------------------------------------------
260         // サブディレクトリ内の同一ファイル名考慮
261         // --------------------------------------------------
262         final String leftDirPath = leftFile.getParentFile().getAbsolutePath();
263         final String rightDirPath = rightFile.getParentFile().getAbsolutePath();
264 
265         // 左右のディレクトリが完全一致する場合は考慮をスキップ
266         if (!leftDirPath.equals(rightDirPath)) {
267             String[] leftDirs = null;
268             String[] rightDirs = null;
269             if ("\\".equals(File.separator)) {
270                 leftDirs = leftDirPath.split("\\\\");
271                 rightDirs = rightDirPath.split("\\\\");
272 
273             } else {
274                 leftDirs = leftDirPath.split("/");
275                 rightDirs = rightDirPath.split("/");
276             }
277 
278             // 左右のディレクトリ階層で不一致になるディレクトリを取得
279             // ※左＝期待値、期待値の流用を想定して、比較対象の右ディレクトリを結果ファイル名に設定
280             boolean isHitDiff = false;
281             int maxLength = 0;
282             if (leftDirs.length < rightDirs.length) {
283                 maxLength = rightDirs.length;
284             } else {
285                 maxLength = leftDirs.length;
286             }
287 
288             for (int idx = 0; idx < maxLength; idx++) {
289                 if (!isHitDiff && !leftDirs[idx].equals(rightDirs[idx])) {
290                     isHitDiff = true;
291                 }
292                 if (isHitDiff) {
293                     // 不一致になった階層～右の対象ファイル配置ディレクトリまでを出力ファイル名に付与
294                     if (idx < rightDirs.length) {
295                         outputFileNameBuilder.append('_').append(rightDirs[idx]);
296                     } else {
297                         break;
298                     }
299                 }
300             }
301 
302         }
303 
304         // 拡張子を付与
305         outputFileNameBuilder.append('.').append(this.outputExt);
306         return outputFileNameBuilder.toString();
307     }

309     /**
310      * ファイル名が、除外ファイル名正規表現リストの定義にマッチするか否かを返します。
311      *
312      * @param file 対象ファイル
313      * @param ignoreFileRegexList 除外ファイル名正規表現リスト
314      * @return 除外する場合、true
315      */
316     private boolean isIgnore(final File file, final List<String> ignoreFileRegexList) {
317         if (ignoreFileRegexList != null) {
318             for (final String ignoreFileRegex : ignoreFileRegexList) {
319                 try {
320                     final Pattern pattern = Pattern.compile(ignoreFileRegex);
321                     final Matcher matcher = pattern.matcher(file.getName());
322                     log.trace("      ・check regex:" + ignoreFileRegex);
323                     if (matcher.matches()) {
324                         // マッチした場合、返却用レイアウト定義に設定
325                         log.info("      ・[SKIP]正規表現:" + ignoreFileRegex + ", ファイル名:" + file.getName());
326                         return true;
327                     }
328                 } catch (Exception e) {
329                     log.error("ファイル名除外ルールの正規表現の評価でエラーが発生しました。正規表現:" + ignoreFileRegex, e);
330                 }
331             }
332         }
333 
334         return false;
335     }

337     /**
338      * ファイルの存在を比較します。
339      *
340      * @param leftFile 左ファイル
341      * @param rightFile 右ファイル
342      * @return 比較ステータス
343      */
344     private CompareStatus fileExistCheck(final File leftFile, final File rightFile) {
345         // ----------------------------------------
346         // ファイルの存在比較
347         // ----------------------------------------
348         if (leftFile.exists()) {
349             if (!rightFile.exists()) {
350                 // 左ファイルのみ
351                 log.info("      ・[左のみ]" + leftFilePath);
352                 return CompareStatus.LeftOnly;
353             }
354 
355         } else {
356             if (rightFile.exists()) {
357                 // 右ファイルのみ
358                 log.info("      ・[右のみ]" + rightFilePath);
359                 return CompareStatus.RightOnly;
360 
361             } else {
362                 // どちらも存在しない
363                 throw new CompareFilesException(Const.MSGCD_ERROR_COMPARE_FILE_BOTH_NOTEXIST, new Object[] {leftFile, rightFile});
364             }
365         }
366 
367         // ----------------------------------------
368         // 0バイトチェック
369         // ----------------------------------------
370         if (leftFile.length() == 0 && rightFile.length() == 0) {
371             // どちらも0バイトの場合、除外として返却
372             log.info("      ・[除外　]どちらも0バイトです。左ファイル=" + leftFilePath + ", 右ファイル=" + rightFilePath);
373             return CompareStatus.Ignore;
374         }
375 
376         return CompareStatus.OK;
377     }
