19  /**
20   * ファイル単位の比較結果。
21   */
22  @lombok.extern.slf4j.Slf4j
23  public class ImageFileCompareResult extends FileCompareResult {
24  
25      private final String inputExt;
26  
27      /**
28       * コンストラクタ。
29       *
30       * @param leftFilePath 左ファイルパス
31       * @param rightFilePath 右ファイルパス
32       * @param fileLayout ファイルレイアウト
33       * @param outputDirPath 出力ディレクトリパス
34       * @param systemConfig システム設定
35       * @param inputExt ファイル拡張子
36       */
37      public ImageFileCompareResult(
38          final String leftFilePath,
39          final String rightFilePath,
40          final FileLayout fileLayout,
41          final String outputDirPath,
42          final CompareFilesConfig systemConfig,
43          final String inputExt) {
44  
45          super(leftFilePath, rightFilePath, outputDirPath, systemConfig);
46          this.outputExt = "png";
47          this.input.setFileLayout(fileLayout);
48  
49          this.inputExt = inputExt;
50      }
51  
52      /*
53       * (非 Javadoc)
54       * @see me.suwash.tools.comparefiles.sv.domain.compare.file.FileCompareResult#compare()
55       */
56      @Override
57      public void compare() {
58          // --------------------------------------------------------------------------------
59          // 事前処理
60          // --------------------------------------------------------------------------------
61          // ----------------------------------------
62          // 単項目チェック
63          // ----------------------------------------
64          ValidateUtils.validate(input);
65  
66          // --------------------------------------------------------------------------------
67          // 本処理
68          // --------------------------------------------------------------------------------
69          // ----------------------------------------
70          // 設定
71          // ----------------------------------------
72          // システム設定
73          final CompareFilesConfig systemConfig = input.getSystemConfig();
74  
75          // 左ファイル
76          final String leftFilePath = input.getLeftFilePath();
77          final File leftFile = new File(leftFilePath);
78          this.leftFilePath = leftFile.getAbsolutePath();
79          // 右ファイル
80          final String rightFilePath = input.getRightFilePath();
81          final File rightFile = new File(rightFilePath);
82          this.rightFilePath = rightFile.getAbsolutePath();
83          // 入力拡張子
84          final String inputExt = this.inputExt;
85          // 出力ファイル
86          final String outputFileName = getOutputFileName(systemConfig);
87          final String outputDirPath = input.getOutputDirPath();
88          final String outputFilePath = outputDirPath + "/" + outputFileName;
89          // 出力拡張子
90          final String outputExt = this.outputExt;
91          // ファイルレイアウト
92          FileLayout fileLayout = input.getFileLayout();
93          if (fileLayout == null) {
94              // 指定されていない場合、画像ファイルのデフォルトを設定
95              fileLayout = FileLayout.getDefaultImageLayout();
96              this.fileLayout = fileLayout;
97          }
98  
99          // 実行開始時刻
100         this.startTime = new Date();
101 
102         // ----------------------------------------
103         // ファイル比較
104         // ----------------------------------------
105         log.info("    ・画像比較");
106         this.status = compare(
107             leftFilePath,
108             rightFilePath,
109             inputExt,
110             outputFilePath,
111             outputExt,
112             fileLayout,
113             systemConfig);
114 
115         // --------------------------------------------------------------------------------
116         // 事後処理
117         // --------------------------------------------------------------------------------
118         // 終了時刻を設定
119         this.endTime = new Date();
120     }
121 
122     /**
123      * 画像を比較します。
124      *
125      * @param leftFilePath 左ファイルパス
126      * @param rightFilePath 右ファイルパス
127      * @param inputExt 入力拡張子
128      * @param outputFilePath 出力ファイルパス
129      * @param outputExt 出力拡張子
130      * @param fileLayout ファイルレイアウト
131      * @param systemConfig システム設定
132      * @return 比較結果
133      */
134     private CompareStatus compare(
135         final String leftFilePath,
136         final String rightFilePath,
137         final String inputExt,
138         final String outputFilePath,
139         final String outputExt,
140         final FileLayout fileLayout,
141         final CompareFilesConfig systemConfig) {
142 
143         // 左ファイル
144         final File leftFile = new File(leftFilePath);
145         // 右ファイル
146         final File rightFile = new File(rightFilePath);
147 
148         // ファイルレイアウトの除外エリアと、システム設定の除外エリアをマージ
149         final List<Rectangle> mergedIgnoreAreaList = new ArrayList<Rectangle>();
150         if (fileLayout != null && fileLayout.getIgnoreAreaList() != null) {
151             mergedIgnoreAreaList.addAll(fileLayout.getIgnoreAreaList());
152         }
153         if (systemConfig.getIgnoreAreaList() != null) {
154             mergedIgnoreAreaList.addAll(systemConfig.getIgnoreAreaList());
155         }
156 
157         // 左ファイル
158         final ImageFileRepository leftRepo = new ImageFileRepository(leftFile, inputExt);
159         leftRepo.begin();
160         final BufferedImage leftImage = leftRepo.next();
161         leftRepo.commit();
162 
163         // 左ファイル
164         final ImageFileRepository rightRepo = new ImageFileRepository(rightFile, inputExt);
165         rightRepo.begin();
166         final BufferedImage rightImage = rightRepo.next();
167         rightRepo.commit();
168 
169         // 比較
170         final DiffAreas diffAreas = ImageCompareUtils.compare(leftImage, rightImage, mergedIgnoreAreaList);
171 
172         // 結果出力
173         this.ignoreRowCount = mergedIgnoreAreaList.size();
174         this.ngRowCount = diffAreas.size();
175 
176         final ImageFileRepository resultRepo = new ImageFileRepository(new File(outputFilePath), outputExt);
177         resultRepo.begin();
178         CompareStatus comparedStatus = null;
179         if (diffAreas.hasDiff()) {
180             comparedStatus = CompareStatus.NG;
181             // 比較結果を出力
182             final BufferedImage result = ImageCompareUtils.getNgImage(
183                 leftImage,
184                 rightImage,
185                 diffAreas,
186                 systemConfig.getLeftPrefix(),
187                 systemConfig.getRightPrefix(),
188                 systemConfig.getNgImageStyle());
189             resultRepo.write(result);
190 
191         } else {
192             comparedStatus = CompareStatus.OK;
193             if (!systemConfig.isWriteDiffOnly()) {
194                 // 差分なしでも出力する場合、比較結果を出力
195                 final BufferedImage result = ImageCompareUtils.getOkImage(
196                     leftImage,
197                     rightImage,
198                     mergedIgnoreAreaList,
199                     systemConfig.getLeftPrefix(),
200                     systemConfig.getRightPrefix(),
201                     systemConfig.getOkImageStyle());
202                 resultRepo.write(result);
203             }
204 
205         }
206 
207         // ファイル操作を確定
208         resultRepo.commit();
209 
210         // 比較結果を返す
211         return comparedStatus;
212     }
