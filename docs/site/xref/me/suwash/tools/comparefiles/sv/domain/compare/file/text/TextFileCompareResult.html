56      /*
57       * (非 Javadoc)
58       * @see me.suwash.tools.comparefiles.sv.domain.compare.file.FileCompareResult#compare()
59       */
60      @Override
61      public void compare() {
62          // --------------------------------------------------------------------------------
63          // 事前処理
64          // --------------------------------------------------------------------------------
65          // ----------------------------------------
66          // 単項目チェック
67          // ----------------------------------------
68          ValidateUtils.validate(input);
69  
70          // --------------------------------------------------------------------------------
71          // 本処理
72          // --------------------------------------------------------------------------------
73          // ----------------------------------------
74          // 設定
75          // ----------------------------------------
76          // 左ファイル
77          this.leftFilePath = input.getLeftFilePath();
78          final File leftFile = new File(leftFilePath);
79          // 右ファイル
80          this.rightFilePath = input.getRightFilePath();
81          final File rightFile = new File(rightFilePath);
82          // 出力ファイル
83          final CompareFilesConfig systemConfig = input.getSystemConfig();
84          final String outputDirPath = input.getOutputDirPath();
85          // ファイルレイアウト
86          this.fileLayout = input.getFileLayout();
87  
88          // 実行開始時刻
89          this.startTime = new Date();
90  
91          // 文字コード判定
92          // ファイルレイアウト > システムデフォルト > 設定読み込み固定値
93          String charset = null;
94          if (fileLayout != null) {
95              charset = fileLayout.getCharset();
96          }
97          if (StringUtils.isEmpty(charset)) {
98              charset = systemConfig.getDefaultInputCharset();
99          }
100         if (StringUtils.isEmpty(charset)) {
101             charset = Const.CHARSET_DEFAULT_CONFIG;
102         }
103 
104         // ----------------------------------------
105         // ソート
106         // ----------------------------------------
107         log.info("    ・ソート");
108 
109         String leftSortedPath = null;
110         String rightSortedPath = null;
111 
112         // ソートフェーズの実施判定
113         if (isSkipSort(systemConfig)) {
114             log.info("      ・Skip");
115             leftSortedPath = leftFilePath;
116             rightSortedPath = rightFilePath;
117 
118         } else {
119             // 左右のソート処理をFork
120             final ForkJoinPool pool = new ForkJoinPool(2);
121 
122             // 左ファイルのソート
123             log.info("      ・左ファイル:" + leftFilePath);
124             final String leftSortedDir = outputDirPath + "/" + Const.DIRNAME_WORK + "/left";
125             leftSortedPath = leftSortedDir + "/" + leftFile.getName();
126             final SortTask leftSortTask = new SortTask(leftFilePath, leftSortedDir, charset, fileLayout, systemConfig);
127             pool.invoke(leftSortTask);
128 
129             // 右ファイルのソート
130             log.info("      ・右ファイル:" + rightFilePath);
131             final String rightSortedDir = outputDirPath + "/" + Const.DIRNAME_WORK + "/right";
132             rightSortedPath = rightSortedDir + "/" + rightFile.getName();
133             final SortTask rightSortTask = new SortTask(rightFilePath, rightSortedDir, charset, fileLayout, systemConfig);
134             pool.invoke(rightSortTask);
135 
136             // ソート処理をJoin
137             if (ProcessStatus.Failure.equals(leftSortTask.join())) {
138                 throw new CompareFilesException(
139                     Const.ERRORHANDLE,
140                     new Object[] {
141                         this.getClass().getSimpleName() + ".Sort.Left",
142                         "file:" + leftFile.getName()
143                     });
144             }
145             if (ProcessStatus.Failure.equals(rightSortTask.join())) {
146                 throw new CompareFilesException(
147                     Const.ERRORHANDLE,
148                     new Object[] {
149                         this.getClass().getSimpleName() + ".Sort.Right",
150                         "file:" + rightFile.getName()
151                     });
152             }
153         }
154 
155 
156         // ----------------------------------------
157         // ファイル比較
158         // ----------------------------------------
159         log.info("    ・テキスト比較");
160         final String outputFileName = getOutputFileName(systemConfig);
161         final String outputFilePath = outputDirPath + "/" + outputFileName;
162         compare(leftSortedPath, rightSortedPath, charset, outputFilePath, fileLayout, systemConfig);
163 
164         // --------------------------------------------------------------------------------
165         // 事後処理
166         // --------------------------------------------------------------------------------
167         // 終了時刻を設定
168         this.endTime = new Date();
169     }

171     /**
172      * ソートフェーズをスキップするか判断します。
173      *
174      * @param systemConfig システム設定
175      * @return ソートフェーズをスキップする場合、true
176      */
177     private boolean isSkipSort(final CompareFilesConfig systemConfig) {
178         // レイアウトから強制スキップ判定
179         if (fileLayout == null) {
180             // レイアウトなしの場合、スキップ
181             return true;
182         }
183 
184         if (FileFormat.Fixed.equals(fileLayout.getFileFormat()) && LineSp.None.equals(fileLayout.getLineSp())) {
185             // 固定長、改行コードなしの場合、スキップ
186             return true;
187         } else if (FileFormat.Json.equals(fileLayout.getFileFormat())) {
188             // Jsonの場合、スキップ
189             return true;
190         }
191 
192         // レイアウトによる強制スキップ以外の場合、設定を利用
193         return systemConfig.isSorted();
194     }

238         protected ProcessStatus compute() {

248     /**
249      * 比較を実行します。
250      *
251      * @param leftFilePath 左ファイルパス
252      * @param rightFilePath 右ファイルパス
253      * @param charset 文字コード
254      * @param outputFilePath 出力ファイルパス
255      * @param fileLayout ファイルレイアウト
256      * @param systemConfig システム設定
257      */
258     private void compare(final String leftFilePath, final String rightFilePath, final String charset, final String outputFilePath, final FileLayout fileLayout, final CompareFilesConfig systemConfig) {
259         // --------------------------------------------------------------------------------
260         // リポジトリ作成
261         // --------------------------------------------------------------------------------
262         final int csvHeaderRow = systemConfig.getCsvHeaderRow();
263         final int csvDataStartRow = systemConfig.getCsvDataStartRow();
264         final String codeValueForOnlyOneRecordType = systemConfig.getCodeValueForOnlyOneRecordType();
265 
266         // 左ファイルリポジトリ
267         final FileRepository<ComparableRow> leftRepo = new GenericRowReadRepository<ComparableRow>(leftFilePath, charset, fileLayout, csvHeaderRow, csvDataStartRow, codeValueForOnlyOneRecordType);
268         leftRepo.begin();
269 
270         // 右ファイルリポジトリ
271         final FileRepository<ComparableRow> rightRepo = new GenericRowReadRepository<ComparableRow>(rightFilePath, charset, fileLayout, csvHeaderRow, csvDataStartRow, codeValueForOnlyOneRecordType);
272         rightRepo.begin();
273 
274         // 行比較結果リポジトリ
275         final TextFileRowCompareResultRepository rowResultRepo = new TextFileRowCompareResultRepositoryImpl(
276             outputFilePath,
277             systemConfig.getOutputCharset(),
278             null,
279             systemConfig.getChunkSize(),
280             systemConfig.isWriteDiffOnly(),
281             systemConfig.getLeftPrefix(),
282             systemConfig.getRightPrefix()
283             );
284         rowResultRepo.begin();
285 
286         // --------------------------------------------------
287         // 左右ともにEOFまでループして比較
288         // --------------------------------------------------
289         // 左ファイル データ読込み
290         ComparableRow leftCurRow = leftRepo.next();
291         // 右ファイル データ読込み
292         ComparableRow rightCurRow = rightRepo.next();
293 
294         // 対象行0件として処理
295         if (leftCurRow == null && rightCurRow == null) {
296             this.status = CompareStatus.Ignore;
297         }
298 
299         // 左右の現在行が両方EOFに到達するまで（nullになるまで）ループ
300         long compareCount = 0;
301         while (true) {
302             compareCount++;
303             if (compareCount % 1000 == 0) {
304                 long leftRowNum = Const.UNKNOWN_LINE;
305                 long rightRowNum = Const.UNKNOWN_LINE;
306                 if (leftCurRow != null) {
307                     leftRowNum = leftCurRow.getRowNum();
308                 }
309                 if (rightCurRow != null) {
310                     rightRowNum = rightCurRow.getRowNum();
311                 }
312                 log.debug("      ・CompareCount:" + compareCount + ", left:" + leftRowNum + ", right:" + rightRowNum);
313             }
314 
315             // EOFチェック
316             if (leftCurRow == null) {
317                 if (rightCurRow == null) {
318                     // --------------------------------------------------
319                     // 左右両方がEOFに到達した場合
320                     // --------------------------------------------------
321                     // ループを抜ける
322                     log.debug("      ・break");
323                     break;
324 
325                 } else {
326                     // --------------------------------------------------
327                     // 左ファイルがEOFに到達した場合
328                     // --------------------------------------------------
329                     log.trace("  ・[LEFT ]reache EOF");
330                     // 比較結果：右のみ
331                     addFixedRowResult(fileLayout, rowResultRepo, CompareStatus.RightOnly, null, rightCurRow);
332                     // 右ファイル データ読込み
333                     rightCurRow = rightRepo.next();
334                     // 次の比較へ
335                     continue;
336                 }
337 
338             } else {
339                 if (rightCurRow == null) {
340                     // --------------------------------------------------
341                     // 右ファイルがEOFに到達した場合
342                     // --------------------------------------------------
343                     log.trace("  ・[RIGHT]reache EOF");
344                     // 比較結果：左のみ
345                     addFixedRowResult(fileLayout, rowResultRepo, CompareStatus.LeftOnly, leftCurRow, null);
346                     // 左ファイル データ読込み
347                     leftCurRow = leftRepo.next();
348                     // 次の比較へ
349                     continue;
350                 }
351             }
352 
353             // どちらもEOFに到達していない場合、キーを比較
354             final int keyCompareResult = leftCurRow.compareTo(rightCurRow);
355             if (keyCompareResult < 0) {
356                 // --------------------------------------------------
357                 // 左キーの方が小さい場合
358                 // --------------------------------------------------
359                 log.trace("  ・[LEFT ]Only");
360                 // 比較結果：左のみ
361                 addFixedRowResult(fileLayout, rowResultRepo, CompareStatus.LeftOnly, leftCurRow, null);
362                 // 左ファイル データ読込み
363                 leftCurRow = leftRepo.next();
364                 // 次の比較へ
365                 continue;
366 
367             } else if (keyCompareResult > 0) {
368                 // --------------------------------------------------
369                 // 右キーの方が小さい場合
370                 // --------------------------------------------------
371                 log.trace("  ・[RIGHT]Only");
372                 // 比較結果：右のみ
373                 addFixedRowResult(fileLayout, rowResultRepo, CompareStatus.RightOnly, null, rightCurRow);
374                 // 右ファイル データ読込み
375                 rightCurRow = rightRepo.next();
376                 // 次の比較へ
377                 continue;
378 
379             } else {
380                 // --------------------------------------------------
381                 // キーが一致した場合
382                 // --------------------------------------------------
383                 // 全項目を比較
384                 addComparedResult(fileLayout, rowResultRepo, leftCurRow, rightCurRow);
385 
386                 // 左ファイル データ読込み
387                 leftCurRow = leftRepo.next();
388                 // 右ファイル データ読込み
389                 rightCurRow = rightRepo.next();
390                 // 次の比較へ
391                 continue;
392             }
393         }
394 
395         // ファイル操作を確定
396         rowResultRepo.commit();
397         leftRepo.commit();
398         rightRepo.commit();
399     }

442     /**
443      * 行比較結果を追加します。
444      *
445      * @param rowResult 行比較結果
446      * @param rowResultRepo 行比較結果リポジトリ
447      */
448     private void addRow(final TextFileRowCompareResult rowResult, final TextFileRowCompareResultRepository rowResultRepo) {
449         // --------------------------------------------------
450         // サマリー項目の更新
451         // --------------------------------------------------
452         final CompareStatus rowStatus = rowResult.getStatus();
453         // ファイル比較結果
454         updateFileStatus(rowStatus);
455         // 行サマリー項目の更新
456         updateRowSummaryFields(rowStatus);
457 
458         // 行比較結果を出力
459         rowResultRepo.write(rowResult);
460     }

462     /**
463      * 実行時点のファイル比較結果と、現在行の比較結果から、ファイル比較結果を更新します。
464      *
465      * @param rowStatus 行比較ステータス
466      */
467     private void updateFileStatus(final CompareStatus rowStatus) {
468         // 行ステータスからサマリー用ステータス(OK or NG)に変換
469         CompareStatus tempSummaryStatus = null;
470         switch (rowStatus) {
471             case OK:
472             case Ignore:
473                 tempSummaryStatus = CompareStatus.OK;
474                 break;
475             case NG:
476             case LeftOnly:
477             case RightOnly:
478                 tempSummaryStatus = CompareStatus.NG;
479                 break;
480             default:
481                 throw new CompareFilesException(Const.UNSUPPORTED);
482         }
483 
484         // 現在のサマリーステータスの状況に合わせて、今回のステータスを適用
485         switch (status) {
486             case Processing:
487             case OK:
488                 status = tempSummaryStatus;
489                 break;
490             case NG:
491                 // 1件でもNGが存在する場合は、更新なし
492                 break;
493             default:
494                 throw new CompareFilesException(Const.UNSUPPORTED);
495         }
496     }

498     /**
499      * 行単位の集計項目を更新します。
500      *
501      * @param rowStatus 行比較ステータス
502      */
503     private void updateRowSummaryFields(final CompareStatus rowStatus) {
504         rowCount++;
505 
506         switch (rowStatus) {
507             case OK:
508                 okRowCount++;
509                 break;
510             case NG:
511                 ngRowCount++;
512                 break;
513             case Ignore:
514                 ignoreRowCount++;
515                 break;
516             case LeftOnly:
517                 leftOnlyRowCount++;
518                 break;
519             case RightOnly:
520                 rightOnlyRowCount++;
521                 break;
522             default:
523                 throw new CompareFilesException(Const.UNSUPPORTED);
524         }
525     }
